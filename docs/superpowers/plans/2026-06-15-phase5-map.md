# Phase 5 — Map Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** An interactive client-only Leaflet map with a tested game-coordinate transform, a factory-pin layer (drag-relocate + right-click-add) wired to saved factories, a layer toggle panel, a pluggable tile background, and an empty-but-wired resource-node layer.

**Architecture:** A pure `coords.ts` module maps Satisfactory world centimetres ↔ Leaflet `CRS.Simple` pixel coordinates. A lazy, client-only `MapView` renders react-leaflet with an optional tile layer (env-configured) and overlay layers. Factory pins read/write the Convex `location` field; resource nodes read a generated JSON slot that is empty until a licensed dataset is supplied.

**Tech Stack:** Leaflet + react-leaflet (new), TanStack Start/Router (client-only route), Convex, React 19, Zod 4, Vitest 4 + testing-library, Biome.

**Spec:** [docs/superpowers/specs/2026-06-14-phase5-map-design.md](../specs/2026-06-14-phase5-map-design.md)

`src/features/map/coords.ts`:

```ts
import { CANVAS_SIZE, WORLD_BOUNDS } from "./config";

export interface World {
	x: number;
	y: number;
}
export interface Pixel {
	px: number;
	py: number;
}

const spanX = WORLD_BOUNDS.maxX - WORLD_BOUNDS.minX;
const spanY = WORLD_BOUNDS.maxY - WORLD_BOUNDS.minY;

export function worldToPixel({ x, y }: World): Pixel {
	return {
		px: ((x - WORLD_BOUNDS.minX) / spanX) * CANVAS_SIZE,
		py: ((y - WORLD_BOUNDS.minY) / spanY) * CANVAS_SIZE,
	};
}

export function pixelToWorld({ px, py }: Pixel): World {
	return {
		x: (px / CANVAS_SIZE) * spanX + WORLD_BOUNDS.minX,
		y: (py / CANVAS_SIZE) * spanY + WORLD_BOUNDS.minY,
	};
}
```

### Task 7: Factory-pin layer

**Files:**

- [ ] **Step 1: Create the layer**

`src/features/map/FactoryPinsLayer.tsx`:

```tsx
import { Link, useNavigate } from "@tanstack/react-router";
import { useMutation, useQuery } from "convex/react";
import L from "leaflet";
import { Marker, Popup, useMapEvents } from "react-leaflet";
import { efficiency } from "#/features/factories/derive";
import { plannedOutputs } from "#/features/factories/factory-view";
import { api } from "#convex/_generated/api";
import type { Doc } from "#convex/_generated/dataModel";
import { pixelToWorld, worldToPixel } from "./coords";

const pinIcon = L.divIcon({
	className: "",
	html: `<div style="width:14px;height:14px;border-radius:50%;background:var(--sea-ink);border:2px solid var(--surface)"></div>`,
	iconSize: [14, 14],
	iconAnchor: [7, 7],
});

function pinLatLng(factory: Doc<"factories">): [number, number] | undefined {
	if (!factory.location) return undefined;
	const { px, py } = worldToPixel(factory.location);
	return [py, px]; // CRS.Simple: lat = y-pixel, lng = x-pixel
}

export default function FactoryPinsLayer() {
	const factories = useQuery(api.factories.list);
	const update = useMutation(api.factories.update);
	const create = useMutation(api.factories.create);
	const navigate = useNavigate();

	useMapEvents({
		contextmenu: async (e) => {
			const location = pixelToWorld({ px: e.latlng.lng, py: e.latlng.lat });
			const id = await create({
				name: "New factory",
				status: "planned",
				production: { source: "manual", inputs: [], outputs: [], machines: [] },
				location,
			});
			navigate({ to: "/factories/$factoryId", params: { factoryId: id } });
		},
	});

	if (!factories) return null;

	return (
		<>
			{factories.map((factory) => {
				const pos = pinLatLng(factory);
				if (!pos) return null;
				const eff = factory.actuals
					? efficiency(plannedOutputs(factory.production), factory.actuals)
					: undefined;
				return (
					<Marker
						key={factory._id}
						position={pos}
						icon={pinIcon}
						draggable
						eventHandlers={{
							dragend: (e) => {
								const ll = e.target.getLatLng();
								update({
									id: factory._id,
									location: pixelToWorld({ px: ll.lng, py: ll.lat }),
								});
							},
						}}
					>
						<Popup>
							<div className="flex flex-col gap-1">
								<span className="font-semibold">{factory.name}</span>
								<span className="text-xs capitalize">{factory.status}</span>
								{eff && (
									<span className="text-xs">
										Efficiency {Math.round(eff.aggregate * 100)}%
									</span>
								)}
								<Link
									to="/factories/$factoryId"
									params={{ factoryId: factory._id }}
									className="text-xs underline"
								>
									Open factory
								</Link>
							</div>
						</Popup>
					</Marker>
				);
			})}
		</>
	);
}
```

```tsx
import L from "leaflet";
import { Marker, Popup } from "react-leaflet";
import { getItem, listResourceNodes } from "#/data";
import type { ResourceNode } from "#/data/schema";
import { worldToPixel } from "./coords";

const PURITY_COLOR: Record<ResourceNode["purity"], string> = {
	impure: "#b07d4b",
	normal: "#c9a227",
	pure: "#4fb8b2",
};

function nodeIcon(purity: ResourceNode["purity"]) {
	return L.divIcon({
		className: "",
		html: `<div style="width:10px;height:10px;border-radius:50%;background:${PURITY_COLOR[purity]};border:1px solid var(--surface)"></div>`,
		iconSize: [10, 10],
		iconAnchor: [5, 5],
	});
}

export default function ResourceNodesLayer({
	purities,
}: {
	purities: Set<ResourceNode["purity"]>;
}) {
	const nodes = listResourceNodes().filter((n) => purities.has(n.purity));
	return (
		<>
			{nodes.map((node) => {
				const { px, py } = worldToPixel(node);
				const item = getItem(node.type);
				return (
					<Marker
						key={node.id}
						position={[py, px]}
						icon={nodeIcon(node.purity)}
					>
						<Popup>
							<span className="text-xs capitalize">
								{item?.name ?? node.type} — {node.purity}
							</span>
						</Popup>
					</Marker>
				);
			})}
		</>
	);
}
```
